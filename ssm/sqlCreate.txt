create table user (
	id varchar(11) not null,
	name varchar(30) not null,
	pwd varchar(30) not null,
	birthday varchar(30),
	gender varchar(30) not null,
	primary key(id)
)charset=utf8;

create table role (
	id varchar(11) not null,
	name varchar(30) not null,
	description varchar(50),
	primary key(id)
)charset=utf8;

create table group1 (
	id varchar(11) not null,
	name varchar(11) not null,
	charge varchar(11) not null,
	primary key(id)
)charset=utf8;

create table user_group (
	id varchar(11) not null,
	user_id varchar(11) not null,
	group_id varchar(11) not null,
	primary key(id)
)charset=utf8;

create table user_role (
	id varchar(11) not null,
	user_id varchar(11) not null,
	role_id varchar(11) not null,
	primary key(id)
)charset=utf8;

create table privelege (
	id varchar(11) not null,
	name varchar(30) not null,
	description varchar(50),
	primary key(id)
)charset=utf8;

create table pri_role (
	id varchar(11) not null,
	role_id varchar(11) not null,
	pri_id varchar(11) not null,
	primary key(id)
)charset=utf8;

create table pri_group (
	id varchar(11) not null,
	group_id varchar(11) not null,
	pri_id varchar(11) not null,
	primary key(id)
)charset=utf8;

create table assignment (
	id varchar(11) not null,
	title varchar(50) not null,
	body varchar(255) not null,
	introducer varchar(30) not null,
	primary key(id)
)charset=utf8;

create table job (
	id varchar(11) not null,
	assign_id varchar(11) not null,
	group_id varchar(11) not null,
	title varchar(50) not null,
	body varchar(255) not null,
	primary key(id)
)charset=utf8;
